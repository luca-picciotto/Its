// area e circonferenza cerchio
#include <iostream>

using namespace std;

int main()
{
    double r, pg = 3.14, A, C;
    cout<<"Inserisci il raggio: ";
    cin >> r;
    
    if(r > 0){
        C = 2 * pg * r;
        A = pg * (r*r);
        cout <<"L'area vale: " <<A <<"\n" <<"La circonferenza vale: " <<C <<endl;
    }
    else{
        cout <<"Errore" <<endl;
    }

    return 0;
}

#include <iostream>

using namespace std;

int main()
{
    double t, t_scontato;
    cout<<"Inserisci totale spesa: ";
    cin >> t;
    
    if(t >= 100){
        if(t >= 300){
            t_scontato = t - (t / 100) * 10;
        }
        else{
            t_scontato = t - (t / 100) * 5;            
        }
    }
    else{
        cout <<"Nessuno sconto per te!"; 
        return 0;
    }
    cout <<"grazie allo sconto sei passato da: " <<t <<" a: " <<t_scontato <<endl;
    return 0;
}
